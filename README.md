# TASK-3

/* 1. View all records */
SELECT *
FROM orders;


/* 2. Total number of orders */
SELECT COUNT(OrderID) AS total_orders
FROM orders;


/* 3. Total revenue generated */
SELECT SUM(TotalPrice) AS total_revenue
FROM orders;


/* 4. Average order value */
SELECT AVG(TotalPrice) AS average_order_value
FROM orders;


/* 5. Orders grouped by status */
SELECT OrderStatus, COUNT(*) AS order_count
FROM orders
GROUP BY OrderStatus
ORDER BY order_count DESC;


/* 6. Top 5 best-selling products (by quantity) */
SELECT Product, SUM(Quantity) AS total_units_sold
FROM orders
GROUP BY Product
ORDER BY total_units_sold DESC
LIMIT 5;


/* 7. Revenue generated by each product */
SELECT Product, SUM(TotalPrice) AS product_revenue
FROM orders
GROUP BY Product
ORDER BY product_revenue DESC;


/* 8. Most frequently used payment methods */
SELECT PaymentMethod, COUNT(*) AS usage_count
FROM orders
GROUP BY PaymentMethod
ORDER BY usage_count DESC;


/* 9. Coupon usage analysis */
SELECT CouponCode, COUNT(*) AS coupon_usage
FROM orders
WHERE CouponCode IS NOT NULL
GROUP BY CouponCode
ORDER BY coupon_usage DESC;


/* 10. Orders by referral source */
SELECT ReferralSource, COUNT(*) AS total_orders
FROM orders
GROUP BY ReferralSource
ORDER BY total_orders DESC;


/* 11. Orders pending delivery */
SELECT *
FROM orders
WHERE OrderStatus = 'Pending';


/* 12. Delivered orders revenue */
SELECT SUM(TotalPrice) AS delivered_revenue
FROM orders
WHERE OrderStatus = 'Delivered';


/* 13. Customers with more than one order */
SELECT CustomerID, COUNT(OrderID) AS orders_count
FROM orders
GROUP BY CustomerID
HAVING COUNT(OrderID) > 1;


/* 14. Highest value orders */
SELECT OrderID, CustomerID, TotalPrice
FROM orders
ORDER BY TotalPrice DESC
LIMIT 10;


/* 15. Average quantity ordered per product */
SELECT Product, AVG(Quantity) AS avg_quantity
FROM orders
GROUP BY Product
ORDER BY avg_quantity DESC;
